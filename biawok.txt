    {"UpperTorso", "Head"},
    {"UpperTorso", "LowerTorso"},
    {"UpperTorso", "RightUpperArm"},
    {"RightUpperArm", "RightLowerArm"},
    {"RightLowerArm", "RightHand"},

local R6_BONES = {
    {"Torso", "Head"},
    {"Torso", "Right Arm"},
    {"Torso", "Left Arm"},
    {"Torso", "Right Leg"},
    {"Torso", "Left Leg"},
}

local REQUIRED_PARTS_R15 = {
    "RightUpperArm", "LeftUpperArm",
    "RightUpperLeg", "LeftUpperLeg",
    "Head", "HumanoidRootPart"
}

local REQUIRED_PARTS_R6 = {
    "Right Arm", "Left Arm",
    "Right Leg", "Left Leg",
    "Head", "Torso"
}

local function safeCreateDrawing(drawType, props)
    local success, obj = pcall(function()
        local d = Drawing.new(drawType)
        if props then
            for k, v in pairs(props) do
                d[k] = v
            end
        end
        return d
    end)
    return success and obj or nil
end

local function createLine()
    return safeCreateDrawing("Line", {
        Visible = false,
        Color = Settings.ESP_Color,
        Thickness = Settings.ESP_Thickness,
        Transparency = Settings.ESP_Transparency
    })
end

local function createSquare()
    return safeCreateDrawing("Square", {
        Visible = false,
        Color = Settings.ESP_BoxColor,
        Thickness = Settings.ESP_Thickness,
        Transparency = Settings.ESP_Transparency,
        Filled = false
    })
end

local function createText()
    return safeCreateDrawing("Text", {
        Visible = false,
        Color = Color3.fromRGB(255, 255, 255),
        Size = 12,
        Center = true,
        Outline = true,
        OutlineColor = Color3.fromRGB(0, 0, 0),
        Font = 1
    })
end

local function createFOVCircle()
    return safeCreateDrawing("Circle", {
        Visible = false,
        Color = Color3.fromRGB(255, 0, 255),
        Thickness = 1.5,
        Transparency = 0.5,
        Filled = false,
        Radius = Settings.Aimbot_FOV,
        Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2),
        NumSides = 64
    })
end

local function hasRequiredLimbs(model)
    if not model then return false end
    local humanoid = model:FindFirstChildOfClass("Humanoid")
    if not humanoid then return false end
    local isR15 = (humanoid.RigType == Enum.HumanoidRigType.R15)
    local requiredParts = isR15 and REQUIRED_PARTS_R15 or REQUIRED_PARTS_R6
    local allFound = true
    for _, partName in ipairs(requiredParts) do
        if not model:FindFirstChild(partName) then
            allFound = false
            break
        end
    end
    return allFound
end

local function getModelBounds(model)
    if not model then return nil end
    local minX, minY, minZ = math.huge, math.huge, math.huge
    local maxX, maxY, maxZ = -math.huge, -math.huge, -math.huge
    local found = false
    for _, part in ipairs(model:GetDescendants()) do
        if part:IsA("BasePart") then
            local pos = part.Position
            local size = part.Size
            local half = size / 2
            local corners = {
                Vector3.new(-half.X, -half.Y, -half.Z),
                Vector3.new( half.X, -half.Y, -half.Z),
                Vector3.new(-half.X,  half.Y, -half.Z),
                Vector3.new( half.X,  half.Y, -half.Z),
                Vector3.new(-half.X, -half.Y,  half.Z),
                Vector3.new( half.X, -half.Y,  half.Z),
                Vector3.new(-half.X,  half.Y,  half.Z),
                Vector3.new( half.X,  half.Y,  half.Z),
            }
            for _, corner in ipairs(corners) do
                local worldPos = pos + corner
                minX = math.min(minX, worldPos.X)
                minY = math.min(minY, worldPos.Y)
                minZ = math.min(minZ, worldPos.Z)
                maxX = math.max(maxX, worldPos.X)
                maxY = math.max(maxY, worldPos.Y)
                maxZ = math.max(maxZ, worldPos.Z)
                found = true
            end
        end
    end
    if not found then return nil end
    return {
        Center = Vector3.new((minX + maxX) / 2, (minY + maxY) / 2, (minZ + maxZ) / 2),
        Size = Vector3.new(maxX - minX, maxY - minY, maxZ - minZ),
        Min = Vector3.new(minX, minY, minZ),
        Max = Vector3.new(maxX, maxY, maxZ),
        Height = maxY - minY,
        Width = maxX - minX,
    }
end

local function addESP(model)
    if not model or trackedEntities[model] then return end
    if not Settings.ESP_ShowLocal and model == LocalPlayer.Character then return end
    if Settings.ESP_RequireLimbs and not hasRequiredLimbs(model) then return end
    local humanoid = model:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    local isR15 = (humanoid.RigType == Enum.HumanoidRigType.R15)
    local bones = isR15 and R15_BONES or R6_BONES
    local linesData = {}
    for _, connection in ipairs(bones) do
        local line = createLine()
        if line then
            table.insert(linesData, {
                PartA = connection[1],
                PartB = connection[2],
                Line = line
            })
        end
    end
    trackedEntities[model] = {
        Humanoid = humanoid,
        Lines = linesData,
        Box = createSquare(),
        HealthBar = createText(),
        NameText = createText(),
        IsR15 = isR15,
        RootPart = model:FindFirstChild("HumanoidRootPart"),
        Head = model:FindFirstChild("Head"),
        Bounds = nil,
        HasLimbs = true,
    }
end

local function removeESP(model)
    if not model or not trackedEntities[model] then return end
    local data = trackedEntities[model]
    if data then
        for _, lineData in ipairs(data.Lines or {}) do
            if lineData and lineData.Line then
                pcall(function() lineData.Line:Remove() end)
            end
        end
        if data.Box then pcall(function() data.Box:Remove() end) end
        if data.HealthBar then pcall(function() data.HealthBar:Remove() end) end
        if data.NameText then pcall(function() data.NameText:Remove() end) end
    end
    trackedEntities[model] = nil
end

local function getClosestPlayerToCrosshair()
    local closest = nil
    local closestDist = Settings.Aimbot_FOV
    for model, data in pairs(trackedEntities) do
        if not model or not data then continue end
        if model == LocalPlayer.Character then continue end
        if not data.Humanoid or data.Humanoid.Health <= 0 then continue end
        if not data.Head then continue end
        if Settings.ESP_RequireLimbs and not hasRequiredLimbs(model) then continue end
        if not Settings.ESP_ShowTeam then
            if LocalPlayer.Team and data.Humanoid:FindFirstChild("Team") then
                local teamValue = data.Humanoid:FindFirstChild("Team")
                if teamValue and LocalPlayer.Team == teamValue.Value then
                    continue
                end
            end
        end
        local headPos = data.Head.Position
        local screenPos, isVisible = Camera:WorldToViewportPoint(headPos)
        if isVisible or not Settings.Aimbot_VisibleCheck then
            local centerX = Camera.ViewportSize.X / 2
            local centerY = Camera.ViewportSize.Y / 2
            local dx = screenPos.X - centerX
            local dy = screenPos.Y - centerY
            local dist = math.sqrt(dx*dx + dy*dy)
            if dist < closestDist then
                closestDist = dist
                closest = {
                    Model = model,
                    Head = data.Head,
                    RootPart = data.RootPart,
                    Humanoid = data.Humanoid,
                    ScreenPos = screenPos,
                    Distance = dist
                }
            end
        end
    end
    return closest
end

local function aimbot()
    if not Settings.Aimbot_Enabled then return end
    local keyName = Settings.Aimbot_Key
    local inputType = getKeyEnum(keyName)
    local isDown = false
    if inputType then
        pcall(function()
            isDown = UserInputService:IsMouseButtonPressed(inputType)
        end)
    end
    if not isDown then return end
    local target = getClosestPlayerToCrosshair()
    if not target then return end
    local headPos = target.Head.Position
    if Settings.Aimbot_Prediction then
        local velocity = target.Head.Velocity
        headPos = headPos + velocity * 0.15
    end
    local screenPos, isVisible = Camera:WorldToViewportPoint(headPos)
    if not isVisible and Settings.Aimbot_VisibleCheck then return end
    local centerX = Camera.ViewportSize.X / 2
    local centerY = Camera.ViewportSize.Y / 2
    local targetX = screenPos.X
    local targetY = screenPos.Y + Settings.Aimbot_HeadOffset
    local smoothness = Settings.Aimbot_Smoothness
    local speed = Settings.Aimbot_Speed
    if smoothness > 1 then
        local dx = targetX - centerX
        local dy = targetY - centerY
        local moveX = dx / smoothness * speed
        local moveY = dy / smoothness * speed
        pcall(function() mousemoverel(moveX, moveY) end)
    else
        pcall(function() mousemoverel(targetX - centerX, targetY - centerY) end)
    end
    if Settings.Aimbot_AutoFire then
        local dist = math.sqrt((targetX - centerX)^2 + (targetY - centerY)^2)
        if dist < Settings.Aimbot_FOV then
            pcall(function()
                VirtualUser:ClickButton2(Vector2.new(targetX, targetY))
            end)
        end
    end
end

RunService.RenderStepped:Connect(function()
    if Settings.Aimbot_Enabled then
        if not fovCircle then
            fovCircle = createFOVCircle()
        end
        if fovCircle then
            fovCircle.Visible = true
            fovCircle.Radius = Settings.Aimbot_FOV
            fovCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
        end
    else
        if fovCircle then
            fovCircle.Visible = false
        end
    end
    if not Settings.ESP_Enabled then
        for model, data in pairs(trackedEntities) do
            if data and data.Lines then
                for _, lineData in ipairs(data.Lines) do
                    if lineData and lineData.Line then
                        lineData.Line.Visible = false
                    end
                end
            end
            if data and data.Box then data.Box.Visible = false end
            if data and data.HealthBar then data.HealthBar.Visible = false end
            if data and data.NameText then data.NameText.Visible = false end
        end
        return
    end
    for model, data in pairs(trackedEntities) do
        if not model or not data or not data.RootPart or not data.Humanoid or data.Humanoid.Health <= 0 then
            removeESP(model)
            continue
        end
        if Settings.ESP_RequireLimbs and not hasRequiredLimbs(model) then
            removeESP(model)
            continue
        end
        local rootPos = data.RootPart.Position
        local screenRoot, rootVis = Camera:WorldToViewportPoint(rootPos)
        if not rootVis or screenRoot.Z < 0 then
            if data.Lines then
                for _, lineData in ipairs(data.Lines) do
                    if lineData and lineData.Line then
                        lineData.Line.Visible = false
                    end
                end
            end
            if data.Box then data.Box.Visible = false end
            if data.HealthBar then data.HealthBar.Visible = false end
            if data.NameText then data.NameText.Visible = false end
            continue
        end
        if Settings.ESP_Skeleton and data.Lines then
            for _, lineData in ipairs(data.Lines) do
                local pA = model:FindFirstChild(lineData.PartA)
                local pB = model:FindFirstChild(lineData.PartB)
                local line = lineData.Line
                if line and pA and pB then
                    local posA, visA = Camera:WorldToViewportPoint(pA.Position)
                    local posB, visB = Camera:WorldToViewportPoint(pB.Position)
                    if posA.Z > 0 and posB.Z > 0 then
                        line.From = Vector2.new(posA.X, posA.Y)
                        line.To = Vector2.new(posB.X, posB.Y)
                        line.Visible = true
                        line.Color = Settings.ESP_Color
                        line.Thickness = Settings.ESP_Thickness
                        line.Transparency = Settings.ESP_Transparency
                    else
                        line.Visible = false
                    end
                elseif line then
                    line.Visible = false
                end
            end
        elseif data.Lines then
            for _, lineData in ipairs(data.Lines) do
                if lineData and lineData.Line then
                    lineData.Line.Visible = false
                end
            end
        end
        if Settings.ESP_Box and data.Box then
            if not data.Bounds then
                data.Bounds = getModelBounds(model)
            end
            local bounds = data.Bounds
            local box = data.Box
            if bounds and bounds.Height > 0 then
                local centerPos = bounds.Center
                local size = bounds.Size
                local halfSize = size / 2
                local corners = {
                    centerPos + Vector3.new(-halfSize.X, -halfSize.Y, -halfSize.Z),
                    centerPos + Vector3.new( halfSize.X, -halfSize.Y, -halfSize.Z),
                    centerPos + Vector3.new(-halfSize.X,  halfSize.Y, -halfSize.Z),
                    centerPos + Vector3.new( halfSize.X,  halfSize.Y, -halfSize.Z),
                    centerPos + Vector3.new(-halfSize.X, -halfSize.Y,  halfSize.Z),
                    centerPos + Vector3.new( halfSize.X, -halfSize.Y,  halfSize.Z),
                    centerPos + Vector3.new(-halfSize.X,  halfSize.Y,  halfSize.Z),
                    centerPos + Vector3.new( halfSize.X,  halfSize.Y,  halfSize.Z),
                }
                local screenCorners = {}
                local allVisible = true
                for _, corner in ipairs(corners) do
                    local screenPos, isVisible = Camera:WorldToViewportPoint(corner)
                    if not isVisible or screenPos.Z <= 0 then
                        allVisible = false
                        break
                    end
                    table.insert(screenCorners, Vector2.new(screenPos.X, screenPos.Y))
                end
                if allVisible and #screenCorners == 8 then
                    local minX, minY = math.huge, math.huge
                    local maxX, maxY = -math.huge, -math.huge
                    for _, sc in ipairs(screenCorners) do
                        minX = math.min(minX, sc.X)
                        minY = math.min(minY, sc.Y)
                        maxX = math.max(maxX, sc.X)
                        maxY = math.max(maxY, sc.Y)
                    end
                    local boxWidth = maxX - minX
                    local boxHeight = maxY - minY
                    if boxWidth < 10 then boxWidth = 10 end
                    if boxHeight < 20 then boxHeight = 20 end
                    box.Position = Vector2.new(minX, minY)
                    box.Size = Vector2.new(boxWidth, boxHeight)
                    box.Visible = true
                    box.Color = Settings.ESP_BoxColor
                    box.Thickness = Settings.ESP_Thickness
                    box.Transparency = Settings.ESP_Transparency
                else
                    local headPos = data.Head and data.Head.Position or rootPos
                    local headScreen = Camera:WorldToViewportPoint(headPos)
                    local rootScreen = Camera:WorldToViewportPoint(rootPos)
                    if headScreen.Z > 0 and rootScreen.Z > 0 then
                        local height = math.abs(headScreen.Y - rootScreen.Y) * 2.2
                        local width = height * 0.55
                        if width < 10 then width = 10 end
                        if height < 20 then height = 20 end
                        local centerX = (headScreen.X + rootScreen.X) / 2
                        box.Position = Vector2.new(centerX - width/2, headScreen.Y - height * 0.05)
                        box.Size = Vector2.new(width, height)
                        box.Visible = true
                        box.Color = Settings.ESP_BoxColor
                        box.Thickness = Settings.ESP_Thickness
                        box.Transparency = Settings.ESP_Transparency
                    else
                        box.Visible = false
                    end
                end
            else
                box.Visible = false
            end
        elseif data.Box then
            data.Box.Visible = false
        end
        if Settings.ESP_HealthBar and data.HealthBar then
            local health = data.Humanoid.Health
            local maxHealth = data.Humanoid.MaxHealth
            local healthPercent = health / maxHealth
            local healthText = data.HealthBar
            healthText.Text = string.format("%.0f%%", healthPercent * 100)
            healthText.Position = Vector2.new(screenRoot.X, screenRoot.Y + 20)
            healthText.Visible = true
            healthText.Color = Settings.ESP_HealthColor
            healthText.Size = 14
        elseif data.HealthBar then
            data.HealthBar.Visible = false
        end
        if data.NameText then
            local displayName = "NPC"
            local player = Players:GetPlayerFromCharacter(model)
            if player then
                displayName = player.DisplayName or player.Name
            end
            data.NameText.Text = displayName
            data.NameText.Position = Vector2.new(screenRoot.X, screenRoot.Y - 25)
            data.NameText.Visible = true
            data.NameText.Color = Color3.fromRGB(255, 255, 255)
            data.NameText.Size = 12
        end
    end
    aimbot()
end)

task.spawn(function()
    while task.wait(1) do
        for _, obj in ipairs(Workspace:GetDescendants()) do
            if obj and obj.Name == "HumanoidRootPart" and obj:IsA("BasePart") then
                local model = obj.Parent
                if model and model:IsA("Model") and not trackedEntities[model] then
                    if not Settings.ESP_RequireLimbs or hasRequiredLimbs(model) then
                        addESP(model)
                    end
                end
            end
        end
    end
end)
